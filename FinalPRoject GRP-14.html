# Function to preprocess text
def preprocess_text(text):
    # Check if the text is a string; if not, return an empty string
    if not isinstance(text, str):
        return ""
    # Convert to lowercase
    text = text.lower()
    # Remove special characters and digits
    text = re.sub(r'\d+', '', text)
    text = re.sub(r'\W+', ' ', text)
    # Tokenization
    words = text.split()
    # Remove stopwords
    stop_words = set(stopwords.words('english'))
    words = [word for word in words if word not in stop_words]
    # Stemming
    stemmer = PorterStemmer()
    words = [stemmer.stem(word) for word in words]
    # Join words back to text
    text = ' '.join(words)
    return text

# Function to find similar movies
def find_similar_movies(input_keywords, top_n=10):
    # Preprocess the input keywords
    input_processed = preprocess_text(input_keywords)
    
    # Transform the input keywords using the same TF-IDF vectorizer
    input_vector = tfidf_vectorizer.transform([input_processed])
    
    # Calculate cosine similarity between input and all movies in the dataset
    cosine_similarities = cosine_similarity(input_vector, tfidf_matrix).flatten()
    
    # Get the top_n most similar movies
    similar_indices = cosine_similarities.argsort()[-top_n:][::-1]
    
    # Prepare the results
    similar_movies = data.iloc[similar_indices][['title', 'keywords']]
    similar_movies['similarity_score'] = cosine_similarities[similar_indices]
    
    return similar_movies

def plot_similar_movies(similar_movies):
    # Check if the DataFrame is empty
    if similar_movies.empty:
        print("No similar movies to plot.")
        return
    
    # Ensure the required columns are present
    if 'title' not in similar_movies.columns or 'similarity_score' not in similar_movies.columns:
        raise KeyError("Required columns 'title' or 'similarity_score' are missing from the DataFrame.")
    
    plt.figure(figsize=(10, 6))
    plt.barh(similar_movies['title'], similar_movies['similarity_score'], color='skyblue')

    plt.ylabel('Movie Title')
    plt.title('Top Similar Movies Based on Similarity Score')
    plt.gca().invert_yaxis()  # Invert y-axis to have the highest score on top
    plt.show()

def recommend_movies_by_keywords(user_input, num_recommendations=10):
    # Preprocess the user input
    processed_input = preprocess_text(user_input)
    
    # Transform the input using the same TF-IDF vectorizer
    user_input_vector = tfidf_vectorizer.transform([processed_input])
    
    # Calculate the cosine similarity
    cosine_similarities = cosine_similarity(user_input_vector, tfidf_matrix).flatten()
    
    # Get the top N movie indices
    top_indices = cosine_similarities.argsort()[-num_recommendations:][::-1]


    # Retrieve the top N movie details
    recommended_movies = data.iloc[top_indices][['original_title', 'release_date', 'director','cast', 'runtime', 'tagline']]
    
    return recommended_movies

# Function to recommend movies based on genre input
def recommend_movies_by_genre(selected_genre, num_recommendations=10):
    # Filter the data based on the selected genre
    genre_filtered = data[data['genres'].str.contains(selected_genre, case=False, na=False)]
    
    if genre_filtered.empty:
        return "No movies found for the selected genre."

    
    # Sort by popularity or any other metric and get the top N recommendations
    genre_recommendations = genre_filtered.nlargest(num_recommendations, 'popularity')[['original_title', 'release_date', 'director','cast', 'runtime', 'tagline']]
    
    return genre_recommendations

def plot_movie_data(data, recommendations=None):
    fig, axs = plt.subplots(2, 2, figsize=(16, 12), constrained_layout=True)

    # Plot 1: Most Popular Movies
    popular_movies = data.nlargest(10, 'popularity')
    axs[0, 0].barh(popular_movies['title'], popular_movies['popularity'], color='skyblue')
    axs[0, 0].set_xlabel('Popularity')
    axs[0, 0].set_ylabel('Movie Title')
    axs[0, 0].set_title('Top 10 Most Popular Movies')
    
   # Plot 2: Genre Distribution
    genre_counts = data['genres'].str.split('|', expand=True).stack().value_counts()
    wedges, texts, autotexts = axs[0, 1].pie(
        genre_counts,
        labels=genre_counts.index,
        autopct='%1.1f%%',
        startangle=140,
        colors=plt.cm.Paired(range(len(genre_counts))),
        wedgeprops=dict(width=0.8)
    )
    axs[0, 1].set_title('Genre Distribution')
    # Adjust text properties for better visibility
    for text in texts:
        text.set_fontsize(6)
    for autotext in autotexts:
        autotext.set_fontsize(6)

    # Plot 3: Recommendations (if provided)
    if recommendations is not None:
        if 'title' not in recommendations.columns or 'similarity_score' not in recommendations.columns:
            raise KeyError("Required columns 'title' or 'similarity_score' are missing from the recommendations DataFrame.")
        axs[1, 0].barh(recommendations['title'], recommendations['similarity_score'], color='lightcoral')
        axs[1, 0].set_xlabel('Similarity Score')
        axs[1, 0].set_ylabel('Movie Title')
        axs[1, 0].set_title('Top Recommended Movies')
    else:
        axs[1, 0].axis('off')  # Hide the plot if no recommendations are provided
    
    # Plot 4: Placeholder or additional plot
    axs[1, 1].axis('off')  # Hide this subplot for now

    plt.show()
